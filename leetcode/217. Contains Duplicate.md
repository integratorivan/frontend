# Решение через hashMap
```js
var containsDuplicate = function (nums) {
    const hash = {};
	for (const num of nums) {
		if (hash[num]) {
			hash[num] = hash[num] + 1;
		} else {
			hash[num] = 1;
			}
		}
	return Object.values(hash).some((item) => item > 1);
};
```

# Решение через Set
```javascript
var containsDuplicate = function(nums) {
    const set = new Set(nums)

    if (set.size !== nums.length) return true;

    return false

};
```
